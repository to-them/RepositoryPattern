-- ORACLE Table
CREATE TABLE students(
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Name VARCHAR2(100) NOT NULL,
    Email VARCHAR2(50) NULL,
    Mobile VARCHAR2(50) NULL,
    Gender VARCHAR2(20) NULL,
    DateOfBirth date,
    PRIMARY KEY(Id)
);
